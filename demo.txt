# Setup Demo 
import-module .\Samples\Azure\AzureProvision.psm1 -Force -scope global 
$startingTime = Get-Date

# Get the existing VM for the demo
$global:vm = Get-AzureDemoVm 

# Open the web site
Write-Host "$($global:vm.DNSName)"

# Update the vm with the DSC Configuration
Update-AzureDemoWordpressVm -vm $global:vm -password 'Pa$$word'

# Get the status of the configuration on the VM
Wait-AzureVmDscExtension -vm $global:vm -startingTime $startingTime

# Open the web site
Write-Host "$($global:vm.DNSName)"
